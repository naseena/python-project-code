def Credit_Card_Offer(salary,Credit_limit):
    if salary<10000:
        Credit_limit=salary*10
    elif  salary>=10000 and salary<=30000:
        Credit_limit=salary*20
    elif salary>30000:
        Credit_limit=salary*30
    return Credit_limit
